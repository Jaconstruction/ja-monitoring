// Gagamit tayo ng alternatibong imbakan na mas mabilis mag-sync const API_URL = "https://api.restful-api.dev/objects/jaconstruction_logs_2024"; let sec = 0; let timer; let isWorking = false; // Clock setInterval(() => { document.getElementById('clock').innerText = new Date().toLocaleTimeString(); }, 1000); // Auto-load logs setInterval(loadLogs, 5000); window.onload = loadLogs; async function loadLogs() { try { const res = await fetch(API_URL); const data = await res.json(); // Dito natin kukunin ang listahan const logs = JSON.parse(data.data.logString || "[]"); displayLogs(logs); document.getElementById('syncStatus').innerText = "Live Master Sync Active"; } catch (e) { document.getElementById('syncStatus').innerText = "Syncing with local backup..."; } } function displayLogs(logs) { const list = document.getElementById('logList'); list.innerHTML = logs.slice().reverse().map(l => `
${l.name}: ${l.action} ${l.time}
`).join(''); } async function handleTime() { const name = document.getElementById('opName').value; if (!name) return alert("Enter operator name!"); const btn = document.getElementById('actionBtn'); if (!isWorking) { isWorking = true; btn.innerText = "TIME OUT"; btn.className = "btn btn-out"; await saveToCloud("TIME IN", name.toUpperCase()); timer = setInterval(() => { sec++; let h = Math.floor(sec/3600).toString().padStart(2,'0'); let m = Math.floor((sec%3600)/60).toString().padStart(2,'0'); let s = (sec%60).toString().padStart(2,'0'); document.getElementById('mainTimer').innerText = `${h}:${m}:${s}`; }, 1000); } else { isWorking = false; clearInterval(timer); await saveToCloud("TIME OUT", name.toUpperCase(), document.getElementById('mainTimer').innerText); btn.innerText = "TIME IN"; btn.className = "btn btn-in"; document.getElementById('mainTimer').innerText = "00:00:00"; sec = 0; } } async function saveToCloud(action, name, dur = "") { const now = new Date(); const newEntry = { date: now.toLocaleDateString(), time: now.toLocaleTimeString(), name, action, duration: dur || 'N/A' }; try { // Kunin ang current logs const res = await fetch(API_URL); const current = await res.json(); let logs = JSON.parse(current.data.logString || "[]"); // Dagdag ang bago logs.push(newEntry); // I-update ang cloud await fetch(API_URL, { method: 'PUT', headers: { 'Content-Type': 'application/json' }, body: JSON.stringify({ name: "JA_Logs", data: { logString: JSON.stringify(logs) } }) }); loadLogs(); } catch (e) { // Backup: Save locally kung mahina net let localLogs = JSON.parse(localStorage.getItem('ja_backup') || "[]"); localLogs.push(newEntry); localStorage.setItem('ja_backup', JSON.stringify(localLogs)); displayLogs(localLogs); } } async function extractLogs() { try { const res = await fetch(API_URL); const data = await res.json(); const logs = JSON.parse(data.data.logString || "[]"); let content = "DATE | TIME | OPERATOR | ACTION | DURATION\n" + logs.map(l => `${l.date} | ${l.time} | ${l.name} | ${l.action} | ${l.duration}`).join('\n'); const blob = new Blob([content], { type: 'text/plain' }); const a = document.createElement('a'); a.href = URL.createObjectURL(blob); a.download = `JA_Master_Logs.txt`; a.click(); } catch (e) { alert("Walang makuha sa cloud."); } }
